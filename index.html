// simple event emitter (TypeScript style)
class Emitter {
  private handlers: Partial void>>> = {};

  on(event: T, handler: (...args: any[]) => void) {
    (this.handlers[event] ||= []).push(handler);
  }

  emit(event: T, ...args: any[]) {
    (this.handlers[event] || []).forEach(h => h(...args));
  }
}
